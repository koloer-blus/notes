# 如何实现？

通过Do函数来控制结构体的执行，只要出现错误我们就捕获进入catch函数，否则执行完成try

type Block struct {  
 Try func()  
 Catch func(interface{})  
 Finally func()  
}  
  
func (t Block) Do() {  
 if t.Finally != nil {  
  defer t.Finally()  
 }  
 if t.Catch != nil {  
  defer func() {  
   if r := recover(); r != nil {  
    t.Catch(r)  
   }  
  }()  
 }  
 t.Try()  
}
